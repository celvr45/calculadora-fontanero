<input id="horas" type="number">

<label>Precio por hora (€)</label>

<input id="precioHora" type="number">

<label>Desplazamiento (€)</label>

<input id="desplazamiento" type="number">

<button onclick="calcular()">Calcular precio</button>

<div class="resultado" id="resultado"></div>

<script>

function calcular(){

let horas = document.getElementById("horas").value

let precioHora = document.getElementById("precioHora").value

let desplazamiento = document.getElementById("desplazamiento").value

let trabajo = document.getElementById("trabajo").value

let manoObra = horas * precioHora

let total = (manoObra * trabajo) + Number(desplazamiento)

document.getElementById("resultado").innerText =
"Precio estimado: " + total + " €"

}

</script>
